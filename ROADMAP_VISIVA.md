def _calculate_ma_signal(self, prices: pd.Series) -> float:
    ma_short = prices.rolling(self.lookback_period // 2).mean()
    ma_long = prices.rolling(self.lookback_period).mean()
    return (ma_short.iloc[-1] / ma_long.iloc[-1]) - 1

def _calculate_volume_ratio(self, volume: pd.Series) -> float:
    avg_volume = volume.rolling(self.lookback_period).mean()
    return volume.iloc[-1] / avg_volume.iloc[-1]
